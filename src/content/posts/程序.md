# 1 弹窗祝福
~~~python
import tkinter as tk
import random

message = [
    "保持好心情，事事顺利！",
    "愿你每天都充满阳光和快乐！",
    "祝你健康平安，幸福美满！",
    "愿你梦想成真，前程似锦！",
    "想你了！希望我们能尽快见面！",
    "你值得拥有最好的！祝福你！",
    "愿你每天都充满爱和温暖！",
    "生活很苦，但你是甜的！",
    "遇见你是我最大的幸福",
    "I LIKE YOU",
]

# 可选背景颜色（暖色系或你喜欢的颜色）
bg_color = ["#F8D7D1", "#F6C7B6", "#F5E0C3", "#F2D6C9", "#FBE8D3", "#F7C9C0"]

POPUP_COUNT = 100
GENERATE_INTERVAL = 80
POPUP_DURATION_MIN = 3000
POPUP_DURATION_MAX = 6000


def create_popup(root):
    try:
        msg = random.choice(message)
        bg = random.choice(bg_color)
        font_size = random.randint(12, 20)

        popup = tk.Toplevel(root)
        popup.title("lmlove")
        popup.configure(bg=bg)

        width = random.randint(150, 220)
        height = random.randint(45, 55)

        screen_width = root.winfo_screenwidth()
        screen_height = root.winfo_screenheight()

        margin = 10
        max_x = max(margin, screen_width - width - margin)
        max_y = max(margin, screen_height - height - margin)
        x = random.randint(margin, max_x)
        y = random.randint(margin, max_y)

        popup.geometry(f"{width}x{height}+{x}+{y}")

        def hex_to_rgb(h):
            h = h.lstrip('#')
            return tuple(int(h[i:i+2], 16) for i in (0, 2, 4))

        def luminance(rgb):
            r, g, b = rgb
            return 0.2126 * r + 0.7152 * g + 0.0722 * b

        def choose_text_color(bg_hex):
            return "#222222" if luminance(hex_to_rgb(bg_hex)) > 180 else "#FFFFFF"

        text_color = choose_text_color(bg)

        label = tk.Label(
            popup,
            text=msg,
            bg=bg,
            fg=text_color,
            font=("Arial", font_size),
            padx=8,
            pady=6,
            wraplength=width - 10,
            justify=tk.CENTER,
        )
        label.pack(expand=True, fill=tk.BOTH)

        popup.after(random.randint(POPUP_DURATION_MIN, POPUP_DURATION_MAX), popup.destroy)
    except Exception as e:
        print("创建弹窗时出错:", e)


if __name__ == "__main__":
    root = tk.Tk()  # 创建主窗口
    root.withdraw()  # 隐藏主窗口

    for i in range(POPUP_COUNT):
        root.after(i * GENERATE_INTERVAL, create_popup, root)

    root.mainloop()
~~~

~~~html
<script>
    (function(){
        "use strict";

        // ---------- 🌸 完全保留原 Python 设定 ----------
        const message = [
            "保持好心情，事事顺利！",
            "愿你每天都充满阳光和快乐！",
            "祝你健康平安，幸福美满！",
            "愿你梦想成真，前程似锦！",
            "想你了！希望我们能尽快见面！",
            "你值得拥有最好的！祝福你！",
            "愿你每天都充满爱和温暖！",
            "生活很苦，但你是甜的！",
            "遇见你是我最大的幸福",
            "I LIKE YOU",
            "I LOVE YOU",
        ];

        const bg_color = ["#F8D7D1", "#F6C7B6", "#F5E0C3", "#F2D6C9", "#FBE8D3", "#F7C9C0"];

        // 时间参数 (毫秒)
        const POPUP_DURATION_MIN = 3000;      // 3秒
        const POPUP_DURATION_MAX = 5500;     // 5.5秒
        const GENERATE_INTERVAL = 80;          // 80ms 生成一个
        const MAX_CONCURRENT_POPUPS = 100;     // 防止过多弹窗 (可自行调整)

        // 全局状态
        let activePopups = new Set();           // 当前存在的泡泡
        let generateTimer = null;               // 自动生成定时器ID
        
        const bodyEl = document.body;

        // ---------- 辅助函数：亮度计算与文字颜色 ----------
        function hexToRgb(hex) {
            const h = hex.replace('#', '');
            return {
                r: parseInt(h.substring(0, 2), 16),
                g: parseInt(h.substring(2, 4), 16),
                b: parseInt(h.substring(4, 6), 16)
            };
        }

        function luminance(r, g, b) {
            return 0.2126 * r + 0.7152 * g + 0.0722 * b;
        }

        function chooseTextColor(bgHex) {
            const rgb = hexToRgb(bgHex);
            const lum = luminance(rgb.r, rgb.g, rgb.b);
            return lum > 180 ? "#222222" : "#FFFFFF";
        }

        function randomInt(min, max) {
            return Math.floor(Math.random() * (max - min + 1)) + min;
        }

        // ---------- 移除单个泡泡 (带动画) ----------
        function removePopup(popup, immediate = false) {
            if (!popup || !activePopups.has(popup)) return;
            if (popup.dataset.removing === 'true') return;
            popup.dataset.removing = 'true';

            const doRemove = () => {
                if (popup.parentNode) {
                    popup.remove();
                }
                activePopups.delete(popup);
            };

            if (immediate) {
                doRemove();
            } else {
                popup.classList.add('fade-out');
                setTimeout(() => {
                    doRemove();
                }, 180);
            }
        }

        // ---------- 创建祝福泡泡 ----------
        function createPopup() {
            // 若泡泡数量过多，暂停生成 (保护性能)
            if (activePopups.size >= MAX_CONCURRENT_POPUPS) {
                return null;
            }

            // 随机选取内容
            const msg = message[Math.floor(Math.random() * message.length)];
            const bg = bg_color[Math.floor(Math.random() * bg_color.length)];
            const fontSize = randomInt(14, 24);
            const textColor = chooseTextColor(bg);

            // 随机尺寸
            const width = randomInt(140, 260);
            const minHeight = randomInt(50, 80);

            // 计算安全位置
            const viewportW = window.innerWidth;
            const viewportH = window.innerHeight;
            const margin = 12;
            const maxLeft = Math.max(margin, viewportW - width - margin);
            const maxTop = Math.max(margin, viewportH - minHeight - margin);
            const left = randomInt(margin, maxLeft);
            const top = randomInt(margin, maxTop);

            // 创建元素
            const popup = document.createElement('div');
            popup.className = 'bless-popup';
            popup.style.backgroundColor = bg;
            popup.style.color = textColor;
            popup.style.fontSize = fontSize + 'px';
            popup.style.width = width + 'px';
            popup.style.minHeight = minHeight + 'px';
            popup.style.left = left + 'px';
            popup.style.top = top + 'px';
            popup.style.padding = '10px 14px';
            popup.textContent = msg;
            
            popup.dataset.removing = 'false';
            
            // 设置自动消失定时器
            const duration = randomInt(POPUP_DURATION_MIN, POPUP_DURATION_MAX);
            const removalTimer = setTimeout(() => {
                removePopup(popup);
            }, duration);
            popup._removalTimer = removalTimer;

            // 点击泡泡可提前关闭 (轻触消失，趣味交互)
            popup.addEventListener('click', (e) => {
                e.stopPropagation();
                if (popup._removalTimer) {
                    clearTimeout(popup._removalTimer);
                    popup._removalTimer = null;
                }
                removePopup(popup);
            });

            bodyEl.appendChild(popup);
            activePopups.add(popup);
            return popup;
        }

        // ---------- 自动生成循环 ----------
        function scheduleNextPopup() {
            if (generateTimer) {
                clearTimeout(generateTimer);
                generateTimer = null;
            }
            generateTimer = setTimeout(() => {
                createPopup();
                scheduleNextPopup();  // 递归调用，持续生成
            }, GENERATE_INTERVAL);
        }

        // 启动自动生成
        function startAutoGenerate() {
            scheduleNextPopup();
        }

        // ---------- 清理资源 (页面卸载时) ----------
        function cleanup() {
            if (generateTimer) {
                clearTimeout(generateTimer);
                generateTimer = null;
            }
            // 清除所有泡泡的定时器
            activePopups.forEach(p => {
                if (p._removalTimer) {
                    clearTimeout(p._removalTimer);
                }
            });
        }

        // ---------- 初始化 ----------
        function init() {
            startAutoGenerate();
            
            // 页面关闭前清理 (虽然不是必须，但更规范)
            window.addEventListener('beforeunload', cleanup);
            
            // 可选：若窗口大小变化，新泡泡位置会自动适配，现有泡泡不动也无伤大雅
        }

        // 启动一切
        init();

        // 完全自动，没有任何控制面板 —— 纯粹的祝福泡泡 ✨
    })();
</script>

</body>
</html>
~~~
